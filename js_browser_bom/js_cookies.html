    Setting/banking the cookie
  • We will want to store info about the visitor in a variable
  • function setCookie(cookname, cookvalue, expdays) {
      var d = new Date();
      d.setTime(d.getTime() + (expdays*24*60*60*1000));
      var expires = "expires="+d.toUTCString();
      document.cookie = cookname + "=" + cookvalue + "; " + expires;
    }
  • arguments:
  • cookname = cookie
  • cookvalue = cookie value
  • expdays = days from now when cookie will expire
  • All of these are combined to set the cookie
    Retrieve cookie
  • function getCookie(cookname) {
      var name = cookname + "=";
      var ca = document.cookie.split(';');
      for(var i=0; i<ca.length; i++) {
        var c = ca[i];
        while (c.charAt(0)==' ') c = c.substring(1);
        if (c.indexOf(name) == 0) return c.substring(name.length,c.length);
      }
      return "";
    }
  • function getCookie takes 1 param; cookname
  • makes a var name;
  • sets name to concatonate cookname + "=";
  • split document.cookie on semicolons int an array

  • go through each index of ca; ca[i]
